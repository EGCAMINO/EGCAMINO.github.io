- قــســم الــرئيسيـة -

هـو الـقـسـم الـمـسـؤول عـن الـصـفـحـة الـرئـيـسـيــة
وهـم مـجـمـوعــة مـن الأعـضـاء وبـعـض الـمـنـسـقـيـن وقـائـد مـســؤول عـنـهــم يـهـتـمــون بـتـرتـيـب الـصـفـحــة الرئـيـسـيــة وطـريـقـة دخـول الـمـدونـات إلـيـهــا
فـيـقـوم الأعـضـاء بـإرسـال مـدونـاتـهـم فـي مـدونـة خـاصـة تـشـرح فـيـهـا كـل شـيء عـن مـعـايـيـر الـقــبـول
ومـن ثـم يـتـم تـقـيـمـهـا مـن قـبـل أعـضـاء قـســم الـرئـيـسـيـة

- أســاســيــات الـتـقـيـــم -

 فـكـرة الـمـدونـة -
 الـمـحـتـوى -
 الـتـنـسـيــق -
 طـول الـمـدونــة -
الـجـــودة -
وعـنـد الـتـأكـد مـن هــذه كـلـهـا يـتـم قـبــول الـمـدونـة اذا تـوفـرت فـيـهـا كـل هـذه الــشـروط
وتـرفض عـنـد عـدم وجـود احـد هـذه الـشــروط
